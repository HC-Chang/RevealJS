Operation - 1


# 複製 github 程式碼
git clone "github example code"

# 新建 git 資料庫
git init

# 增加追蹤檔案
git add.
# 確認追蹤檔案並添加註解
git commit -m "Comment"

# 查看 git 紀錄
git log
# 簡易 git log 紀錄圖
git log --oneline --decorate --graph --all

Operation - 2


# 建立分支
git branch "Branch Name"
# 切換至其他分支
git checkout "Branch Name"

# 從雲端資料庫(github)更新至本地資料庫
git pull
# 從本地資料庫更新至雲端資料庫
git push
